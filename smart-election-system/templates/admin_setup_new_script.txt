            <script>
                (function() {
                    const fieldsContainer = document.getElementById('candidate-fields');
                    const addButton = document.getElementById('add-candidate-button');
                    const instNameInput = document.getElementById('institution_name');
                    const adminUsernameInput = document.getElementById('admin_username');
                    const adminPasswordInput = document.getElementById('admin_password');

                    // Helper validation functions
                    function validateAdminUsername(username) {
                        const regex = /^[a-zA-Z]{3,}[_][a-zA-Z]{1,}$/;
                        return regex.test(username);
                    }

                    function validateNumericPassword(password, minDigits) {
                        return /^\d+$/.test(password) && password.length >= minDigits;
                    }

                    function validateCandidateName(name) {
                        return /^[a-zA-Z]{3,}$/.test(name);
                    }

                    function showError(elementId, message) {
                        const errorEl = document.getElementById(elementId);
                        if (errorEl) {
                            errorEl.textContent = message;
                            errorEl.style.display = 'block';
                        }
                    }

                    function hideError(elementId) {
                        const errorEl = document.getElementById(elementId);
                        if (errorEl) {
                            errorEl.style.display = 'none';
                        }
                    }

                    // Real-time validation for institution name
                    instNameInput.addEventListener('input', function() {
                        const value = this.value.trim();
                        if (value && value.length < 3) {
                            showError('err_institution_name', 'Enter a valid name (minimum 3 characters)');
                        } else {
                            hideError('err_institution_name');
                        }
                    });

                    // Real-time validation for admin username
                    adminUsernameInput.addEventListener('input', function() {
                        const value = this.value.trim();
                        if (value && !validateAdminUsername(value)) {
                            showError('err_admin_username', 'Use format: firstname_secondname (e.g., sam_k, real name only)');
                        } else {
                            hideError('err_admin_username');
                        }
                    });

                    // Real-time validation for admin password
                    adminPasswordInput.addEventListener('input', function() {
                        const value = this.value;
                        if (value && !validateNumericPassword(value, 4)) {
                            showError('err_admin_password', 'Password must be at least 4 digits (numbers only)');
                        } else {
                            hideError('err_admin_password');
                        }
                    });

                    function getCandidateCount() {
                        return fieldsContainer.querySelectorAll('.candidate-row').length;
                    }

                    function buildLabelText(index) {
                        return `CANDIDATE ${index}${index <= 2 ? ' (REQUIRED)' : ''}`;
                    }

                    function createCandidateRow(index) {
                        const row = document.createElement('div');
                        row.className = 'candidate-row';
                        row.style.display = 'flex';
                        row.style.gap = '12px';
                        row.style.alignItems = 'flex-end';

                        const group = document.createElement('div');
                        group.className = 'input-group';
                        group.style.flex = '1';

                        const label = document.createElement('label');
                        label.style.display = 'block';
                        label.style.fontSize = '0.75rem';
                        label.style.fontWeight = '600';
                        label.style.color = 'var(--text-secondary)';
                        label.style.marginBottom = '8px';
                        label.textContent = buildLabelText(index);

                        const input = document.createElement('input');
                        input.type = 'text';
                        input.name = `candidate${index}`;
                        input.id = `candidate${index}`;
                        input.placeholder = 'e.g., sam';
                        input.required = true;
                        input.style.width = '100%';
                        input.style.background = 'rgba(0,0,0,0.2)';
                        input.style.border = '1px solid var(--border-glass)';
                        input.style.borderRadius = '12px';
                        input.style.padding = '14px';
                        input.style.color = 'white';
                        input.style.outline = 'none';
                        input.style.transition = '0.3s';
                        input.onfocus = function() { this.style.borderColor = 'var(--accent-violet)'; };
                        input.onblur = function() { this.style.borderColor = 'var(--border-glass)'; };

                        const errorMsg = document.createElement('small');
                        errorMsg.id = `err_candidate${index}`;
                        errorMsg.style.fontSize = '0.7rem';
                        errorMsg.style.color = '#ff6b6b';
                        errorMsg.style.marginTop = '4px';
                        errorMsg.style.display = 'none';

                        // Real-time validation for candidate
                        input.addEventListener('input', function() {
                            const value = this.value.trim();
                            if (value && !validateCandidateName(value)) {
                                showError(errorMsg.id, 'Enter a valid name (min 3 letters, avoid random text like sdf)');
                            } else {
                                hideError(errorMsg.id);
                            }
                        });

                        const removeButton = document.createElement('button');
                        removeButton.type = 'button';
                        removeButton.textContent = '−';
                        removeButton.title = 'Remove candidate';
                        removeButton.style.width = '48px';
                        removeButton.style.height = '48px';
                        removeButton.style.borderRadius = '50%';
                        removeButton.style.border = '1px solid var(--border-glass)';
                        removeButton.style.background = 'rgba(255,255,255,0.05)';
                        removeButton.style.color = 'white';
                        removeButton.style.fontSize = '1.5rem';
                        removeButton.style.cursor = 'pointer';
                        removeButton.addEventListener('click', function() {
                            row.remove();
                            refreshCandidateRows();
                        });

                        group.appendChild(label);
                        group.appendChild(input);
                        group.appendChild(errorMsg);
                        row.appendChild(group);
                        row.appendChild(removeButton);
                        return row;
                    }

                    function refreshCandidateRows() {
                        const rows = Array.from(fieldsContainer.querySelectorAll('.candidate-row'));
                        rows.forEach((row, index) => {
                            const idx = index + 1;
                            const label = row.querySelector('label');
                            const input = row.querySelector('input');
                            if (label && input) {
                                label.textContent = buildLabelText(idx);
                                input.name = `candidate${idx}`;
                                input.id = `candidate${idx}`;
                                input.required = idx <= 2;
                            }
                            const removeButton = row.querySelector('button');
                            if (removeButton) {
                                removeButton.style.display = idx <= 2 ? 'none' : 'inline-flex';
                            }
                        });
                    }

                    addButton.addEventListener('click', function() {
                        const nextIndex = getCandidateCount() + 1;
                        fieldsContainer.appendChild(createCandidateRow(nextIndex));
                    });

                    fieldsContainer.querySelectorAll('.candidate-row').forEach((row, index) => {
                        const removeButton = row.querySelector('button');
                        if (removeButton) {
                            removeButton.style.display = 'none';
                        }
                    });

                    const form = addButton.closest('form');
                    if (form) {
                        form.addEventListener('submit', function(event) {
                            let hasError = false;

                            // Validate institution name
                            const instName = instNameInput.value.trim();
                            if (!instName || instName.length < 3) {
                                showError('err_institution_name', 'Enter a valid name (minimum 3 characters)');
                                hasError = true;
                            } else {
                                hideError('err_institution_name');
                            }

                            // Validate admin username
                            const adminUser = adminUsernameInput.value.trim();
                            if (!adminUser || !validateAdminUsername(adminUser)) {
                                showError('err_admin_username', 'Use format: firstname_secondname (e.g., sam_k, real name only)');
                                hasError = true;
                            } else {
                                hideError('err_admin_username');
                            }

                            // Validate admin password
                            const adminPass = adminPasswordInput.value;
                            if (!adminPass || !validateNumericPassword(adminPass, 4)) {
                                showError('err_admin_password', 'Password must be at least 4 digits (numbers only)');
                                hasError = true;
                            } else {
                                hideError('err_admin_password');
                            }

                            // Validate candidate names
                            const inputs = Array.from(fieldsContainer.querySelectorAll('input[name^="candidate"]'));
                            for (const input of inputs) {
                                const value = input.value.trim();
                                const candNum = input.id.replace('candidate', '');
                                if (!value) {
                                    showError(`err_candidate${candNum}`, 'Fill this field or remove the row');
                                    hasError = true;
                                } else if (!validateCandidateName(value)) {
                                    showError(`err_candidate${candNum}`, 'Enter a valid name (min 3 letters, avoid random text like sdf)');
                                    hasError = true;
                                } else {
                                    hideError(`err_candidate${candNum}`);
                                }
                            }

                            if (hasError) {
                                event.preventDefault();
                            }
                        });
                    }
                })();
            </script>
